[Docker Commands]

# Return the list of all running containers.
# [options]
# -a to view all containers running or not.

docker ps

# To run and install docker image
# [options]
#    -d detach
#    -p port
#    -e environment variable
#    --name container_name
#    --net for docker network

docker run -d -p 80:80 image_name

# To create image
docker build -t "project name" .

# To pull image
docker pull image_name

#to check all docker images in local system
docker images

# to stop container
docker stop container_id

# to start container
docker start container_id

# to see the con container logs
docker logs container_id :

# get the terminal to current container
docker exec -it container_id /bin/bash


Docker Network : if two container deployed in same docker network they can
communicate with just container name.

# to create docker networks
docker network create network-name


[Powershell]
# to install linux distribution in windows linux system(WLS).

wsl -l -v : display list of installed linux distribution.
wsl --install -d Ubuntu : install distribution
wsl --set-default-version 2 : set wsl version 2.

[LINUX- ubuntu]

1. apt-get update : update package index and install packages

[Docker installation]

1. set repository
echo \
  "deb [arch="$(dpkg --print-architecture)" signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  "$(. /etc/os-release && echo "$VERSION_CODENAME")" stable" | \
  sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

2. sudo apt-get update

3.Install docker
sudo apt-get install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

4.Create docker group and add user.

sudo groupadd docker
sudo usermod -aG docker $USER

5. sudo docker run hello-world
